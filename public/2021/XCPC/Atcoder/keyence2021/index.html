using namespace std;

typedef long long ll;
#define maxn 5005
#define mod 998244353
char imap[maxn][maxn];
ll dp[maxn][maxn];
ll qpow(ll a,ll n)
{
	ll ans=1;
	while (n)
	{
		if ( n & (ll)1) ans=ans*a%mod;
		a=a*a%mod;
		n>>=1;
	}
	return ans%mod;
}

int main()
{
	int H,W,K,h,w;
	char k;
	cin>>H>>W>>K;
	for (int i=1;i<=K;i++)		
	{
		scanf("%d %d %c",&h,&w,&k);
		imap[h][w]=k;
	}
	ll inv=qpow(3,mod-2);
	dp[1][1]=1;
	for (int i=1;i<=H;i++)
	{
		for (int j=1;j<=W;j++)
		{
			if (imap[i][j]=='R'||imap[i][j]=='X') 
                dp[i][j+1]=(dp[i][j]+dp[i][j+1] )%mod;
			if (imap[i][j]=='D'||imap[i][j]=='X') 
                dp[i+1][j]=(dp[i][j]+dp[i+1][j] )%mod;

	}

}
