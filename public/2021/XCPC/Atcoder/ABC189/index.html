#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int inf=1e9+7;
const int maxn=2e5+10;
ll A[maxn]={0};
bool exist[maxn];

struct P
{
	long double k,b;
	P operator + (const P& a)
	{
		return {k+a.k,b+a.b};
	}
	
	P operator - (const P& a)
	{
		return {k-a.k,b-a.b};
	}
		
	P operator / (const long double a)
	{
		return {k/a,b/a};
	}
}f[maxn],sum[maxn];

int main(){
	int N,K;
	long double M;
	scanf("%d%Lf%d",&N,&M,&K);
	for (int i=1;i<=K;i++)
	{
		scanf ("%d",&A[i]);
		exist[A[i]]=true;
	} 
	for (int i=N-1;i>=0;i--)
	{
		if (exist[i]) f[i]=(P){1,0};
		else f[i]=(sum[i+1]-sum[int(i+M+1)])/M+(P){0,1};
		sum[i]=sum[i+1]+f[i];
	}
	if (fabs(f[0].k-1)<1e-6) printf("-1\n");
	else printf("%.4Lf\n",-f[0].b/(f[0].k-1));
	return 0;
}
