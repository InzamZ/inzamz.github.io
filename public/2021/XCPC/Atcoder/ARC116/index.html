#include <bits/stdc++.h>
using namespace std;
const int maxn = 2e5 + 10;
using ll = long long;
const ll mod = 998244353;

ll a[maxn], ans = 0;

inline ll qpow(int a, int n, int m)
{
    long long ans = 1;
    while(n){
        if(n&1){
            ans = (ans * a) % m;
        }
        a = (a * a) % m;
        n >>= 1;
    }
    return ans;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        cin >> a[i];
        ans += ((a[i] * a[i]) % mod);
        ans %= mod;
    }
    sort(a, a + n);
    for (int i = 0; i < n; ++i)
    {
        for (int j = i + 1; j < n; ++j)
        {
            ans += ((a[i] * a[j]) % mod * qpow(2,j - i - 1,mod)) % mod;
            ans %= mod;
        }
    }
    cout << ans << endl;
    return 0;
}
