#include <bits/stdc++.h>
int ans[20005][6],top=0;
bool s[110][110];

void prians(int a,int b,int c,int d,int e,int f)
{
        ans[top][0]=a;
        ans[top][1]=b;
        ans[top][2]=c;
        ans[top][3]=d;
        ans[top][4]=e;
        ans[top][5]=f;
        s[a][b]=!s[a][b];
        s[c][d]=!s[c][d];
        s[e][f]=!s[e][f];
        top++;
        return ;
}

int main()
{
        int n,m,t;
        scanf ("%d",&t);
        while (t--){
                top=0;
                scanf ("%d %d ",&n,&m);
                for (int i=1;i<=n;i++){
                        for (int j=1;j<=m;j++){
                                scanf("%1d",&s[i][j]);
                        }
                }
                for (int i=1;i<n-1;i++){
                        for (int j=1;j<m;j++){
                                if (s[i][j]) prians(i,j,i+1,j,i+1,j+1);
