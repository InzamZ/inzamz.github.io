struct com
{
	ll tim,tar;
	com(){}
	com(int a,int b):tim(a),tar(b){}
	bool operator< (const com x) const {
		return tim<x.tim;

{
	return a.tim<b.tim;

{
	int T,n;

	{

		}

		{
			ll t1,t2,L,R;
			if (s[i].tim>=overt) 
			{

				pos=target;
				target=s[i].tar;

				else dir=true;
