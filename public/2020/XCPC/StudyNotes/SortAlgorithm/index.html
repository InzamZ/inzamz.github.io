void MergeArray(int* s,int l,int r,int* temp)
{
	int i=l,m=(l+r)/2,j=m+1,k=l;
	while ( i<=m && j<=r )
	{
		if (s[i]<=s[j]) temp[k]=s[i++];
		else temp[k]=s[j++];
		k++;
	}
	while (i<=m) temp[k]=s[i++],k++;
	while (j<=r) temp[k]=s[j++],k++;
	i=l;
	while (i<=r)
	{
		s[i]=temp[i];
		i++;
	}
}

    int dad=start,son=dad*2+1;
