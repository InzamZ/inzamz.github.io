其实就是一个组合数问题 , 问题是如何使用组合数 .

组合数模板 & 题解

#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

typedef long long ll;
const ll mod=1e9+7;
const int maxn=200010;
ll t,m,n,k,fac[maxn],inv[maxn],cnt[maxn],num[maxn];
ll qpow(ll x,ll y)
{
    ll res=1;
    while (y)
    {
        if (y & 1ll) res=res*x%mod;
        y>>=1;
        x=x*x%mod;
    }
    return res;
}

ll C(ll m,ll n)
{
    if ( n>m || n<0) return 0;
    return (fac[m]*inv[n]%mod)*inv[m-n]%mod;
}

int main()
{
    fac[0]=inv[0]=1;
    for (ll i=1;i<maxn;i++) fac[i]=fac[i-1]*i%mod;
    for (ll i=1;i<maxn;i++) inv[i]=qpow(fac[i],mod-2);
    scanf("%lld",&t);
    while (t--)
    {
        scanf("%lld%lld%lld",&n,&m,&k);
        memset(cnt,0,sizeof(ll)*(n+1));
        for(int i=1;i<=n;i++) {
            scanf("%lld",&num[i]);
            cnt[num[i]]++;
        }
        sort(num+1,num+1+n);
        for(int i=1;i<=n;i++) cnt[i]+=cnt[i-1];
        ll ans=0;
        for (int i=n;i>0;i--)
        {
            if (num[i]>k) ans+=C(i-cnt[num[i]-k-1]-1,m-1),ans%=mod;
            else ans+=C(i-1,m-1),ans%=mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
